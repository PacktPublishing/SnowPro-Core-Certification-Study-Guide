-- Create a single-cluster warehouse using SQL commands:
CREATE OR REPLACE WAREHOUSE test-wh
WITH WAREHOUSE_TYPE = STANDARD
WAREHOUSE_SIZE = XSMALL;

-- Create a multi-cluster warehouse using SQL commands:
CREATE OR REPLACE WAREHOUSE test-wh
WITH WAREHOUSE_TYPE = STANDARD
WAREHOUSE_SIZE = XSMALL
MAX_CLUSTER_COUNT = 4
MIN_CLUSTER_COUNT = 2
SCALING_POLICY = ECONOMY;
